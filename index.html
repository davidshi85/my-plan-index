    const infoContent = document.getElementById('info-content');

    const floorPlanContainer = document.getElementById('floor-plan-container');

    const tabButtons = document.querySelectorAll('.tab-button');

    const tabContentContainer = document.getElementById('tab-content');

    let activeElement = null;

 

    function updateInfoPanel(id = 'default') {

        const data = planData[id] || planData.default;

        infoContent.style.opacity = 0;

        setTimeout(() => {

            infoContent.innerHTML = `

                <h3 class="text-xl font-bold mb-2 flex items-center"><span class="text-2xl mr-3">${data.icon}</span> ${data.title}</h3>

                <div class="prose prose-stone max-w-none">${data.content}</div>

            `;

            infoContent.style.opacity = 1;

        }, 300);

    }

 

    function switchTab(tabId) {

         tabButtons.forEach(btn => {

            btn.classList.remove('active');

            if(btn.dataset.tab === tabId) {

                btn.classList.add('active');

            }

        });

        tabContentContainer.innerHTML = tabData[tabId];

    }

 

    floorPlanContainer.addEventListener('click', (e) => {

        const targetElement = e.target.closest('.plan-element');

 

        if (activeElement) {

            activeElement.classList.remove('active-element');

        }

 

        if (targetElement) {

            const id = targetElement.dataset.id;

            updateInfoPanel(id);

            targetElement.classList.add('active-element');

            activeElement = targetElement;

        } else {

            updateInfoPanel();

            activeElement = null;

        }

    });

 

    tabButtons.forEach(button => {

        button.addEventListener('click', () => {

            switchTab(button.dataset.tab);

        });

    });

 

    document.addEventListener('DOMContentLoaded', () => {

        updateInfoPanel();

        switchTab('tab-furniture');

    });
